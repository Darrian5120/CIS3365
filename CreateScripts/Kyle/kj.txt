CREATE TABLE CUSTOMER_STATUS ( 

  

C_ID INT(9) NOT NULL, 

C_ACTIVE BOOL NOT NULL, 

PRIMARY KEY (C_ID), 

FOREIGN KEY (C_ID) REFERENCES Customer(C_ID) 

  

); 

  

CREATE TABLE SUPPLIER_STATUS( 

  

SUPPLIER_ID INT(9), 

SUPPLIER_STATUS BOOL, 

PRIMARY KEY (C_ID), 

FOREIGN KEY (C_ID) REFERENCES Customer(C_ID) 

  

  

); 

  

CREATE TABLE ACCOUNT_ASSETS( 

  

ACCOUNT_ID INT(9), 

ASSET_NAME VARCHAR(30), 

ASSET_VALUE DOUBLE(10), 

PRIMARY KEY (ACCOUNT_ID) 

  

); 

  

CREATE TABLE ACCOUNT_LIABILITY( 

  

LIABILITY_ID INT(9), 

LIABILITY_NAME VARCHAR(30), 

LIABILITY_VALUE DOUBLE(10), 

PRIMARY KEY (LIABILITY_ID) 

  

  

); 

  

CREATE TABLE VEHICLE( 

  

V_VIN INT(17), 

V_MAKE VARCHAR(15), 

V_LISCENSE_PLATE VARCHAR(10), 

V_MODEL VARCHAR(15), 

V_YEAR INT(4), 

PRIMARY KEY (V_VIN) 

  

);
